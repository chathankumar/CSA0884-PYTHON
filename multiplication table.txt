def print_multiplication_table(m, n):
    for i in range(1, n+1):
        print(f"{m} x {i} = {m*i}")
m = int(input("Enter a number to print its multiplication table: "))
n = int(input("Enter the number of multiples to be printed: "))
print(f"Multiplication table of {m} up to {n} is:")
print_multiplication_table(m, n)
